# usage_example.py
import time
from desktop_manager_core import get_desktop_manager_core

def main():
    # Khởi tạo hệ thống
    desktop_mgr = get_desktop_manager_core()
    
    print("🎯 HỆ THỐNG QUẢN LÝ KỸ NĂNG DESKTOP")
    print("=" * 50)
    
    # 1. Ghi kỹ năng mới
    print("1. 🎥 Bắt đầu ghi kỹ năng...")
    result = desktop_mgr.start_skill_recording({
        "learning_objective": "learn_to_open_browser",
        "target_element": "browser_icon"
    })
    
    if result['success']:
        print(f"   Đang ghi: {result['demo_name']}")
        print("   Thực hiện thao tác của bạn trong 5 giây...")
        time.sleep(5)  # Giả lập thao tác người dùng
        
        # Dừng ghi
        print("2. ⏹️ Dừng ghi và tạo kỹ năng...")
        result = desktop_mgr.stop_skill_recording()
        
        if result['success']:
            print(f"   ✅ Đã tạo kỹ năng: {result['skill_name']}")
            print(f"   📝 Mô tả: {result['skill_description']}")
    
    # 3. Thực thi kỹ năng
    print("3. 🎮 Thực thi kỹ năng...")
    skill_name = result['skill_name']
    execute_result = desktop_mgr.execute_skill(skill_name)
    
    if execute_result['success']:
        print(f"   ✅ Đã thực thi: {skill_name}")
    
    # 4. Khám phá tự động
    print("4. 🔍 Khám phá kỹ năng mới...")
    explore_result = desktop_mgr.explore_and_learn_skills()
    if explore_result['success']:
        print(f"   Đã khám phá {len(explore_result['discovered_skills'])} kỹ năng")
    
    # 5. Liệt kê kỹ năng
    print("5. 📋 Liệt kê kỹ năng...")
    skills = desktop_mgr.list_skills()
    print(f"   Tổng số kỹ năng: {skills['total_skills']}")
    for name, info in skills['skills'].items():
        print(f"   - {name}: {info['description']}")

if __name__ == "__main__":
    main()
